sc1\_ver = [1, 2, 3, 4]

class ColorFormatter(logging.Formatter):
    # 日志级别中文对应及颜色
    COLORS = {
        'DEBUG': Fore.CYAN,
        'INFO': Fore.LIGHTMAGENTA\_EX,
        'WARNING': Fore.YELLOW,
        'ERROR': Fore.RED,
        'CRITICAL': Fore.LIGHTRED\_EX,
    }
    # 日志级别中文翻译
    LEVEL\_NAMES = {
        'DEBUG': '调试',
        'INFO': '信息',
        'WARNING': '警告',
        'ERROR': '错误',
        'CRITICAL': '严重错误'
    }

    def format(self, record):
        color = self.COLORS.get(record.levelname, Style.RESET\_ALL)
        level\_name = self.LEVEL\_NAMES.get(record.levelname, record.levelname)
        return f"{color}[{level\_name}] {record.getMessage()}{Style.RESET\_ALL}"

logger = logging.getLogger()
logger.setLevel(logging.INFO)
handler = logging.StreamHandler()
handler.setFormatter(ColorFormatter())
logger.handlers = [handler]

def verify\_files():
    logger.info("正在验证文件完整性...")
    if not os.path.exists("lib/ScDowngrade.exe"):
        logger.warning("缺少 ScDowngrade.exe")
    if not os.path.exists("lib/SctxConverter.exe"):
        logger.warning("缺少 SctxConverter.exe")

def sc\_file\_filter(path):
    return path.endswith(".sc") and not path.endswith("\_tex.sc")

def print\_centered(text, color\_code=""):
    width = shutil.get\_terminal\_size().columns
    print(f"{color\_code}{text.center(width)}{Style.RESET\_ALL}")

def get\_used\_version(data):
    if len(data) < 6 or data[:2] != b"SC":
        return None
    be = struct.unpack(">I", data[2:6])[0]
    le = struct.unpack("<I", data[2:6])[0]
    if be in sc1\_ver or be in sc2\_ver:
        return be
    if le in sc1\_ver or le in sc2\_ver:
        return le
    return None

def downgrade(filepath):
    try:
        subprocess.run([EXE\_PATH, filepath, filepath], check=True)
        logger.info(f"已降级: {os.path.basename(filepath)}")
    except FileNotFoundError:
        logger.critical(f"在路径 ~\\\\{EXE\_PATH} 找不到 ScDowngrade.exe")
    except subprocess.CalledProcessError:
        logger.error(f"ScDowngrade.exe 处理失败: {os.path.basename(filepath)}")
    except Exception as e:
        logger.error(f"意外错误: {os.path.basename(filepath)} — {e}")

def process\_file(filepath, dump):
    importlib.invalidate\_caches()
    import lib.sc\_import as sc\_import
    importlib.reload(sc\_import)
    sc\_import.DUMP = dump
    from lib import sc\_to\_fla

    print("-" \* 20)
    logger.info(f"正在处理: {os.path.basename(filepath)}")

    with open(filepath, "rb") as f:
        data = f.read()

    version = get\_used\_version(data)
    if version is None:
        logger.critical(f"文件格式错误: {os.path.basename(filepath)}")
        return

    if version in sc1\_ver:
        sc\_to\_fla(filepath)
    elif version in sc2\_ver:
        logger.info("检测到 SC2 格式文件 - 正在降级")
        downgrade(filepath)

        with open(filepath, "rb") as f:
            data = f.read()
        version = get\_used\_version(data)

        if version is not None and version not in sc2\_ver:
            logger.info("正在处理 SC1 格式文件")
            sc\_to\_fla(filepath)
        else:
            logger.warning("处理失败! 跳过文件...")
    else:
        logger.critical(f"不支持的版本: {os.path.basename(filepath)}")

def main():
    verify\_files()
    sort\_layers = False # 图层排序状态

    while True:
        show\_menu()
        try:
            choice = input(f"{Fore.CYAN}请输入操作序号 (1-7): {Style.RESET\_ALL}")

            if choice == "1":
                path = input(f"{Fore.CYAN}请输入文件或目录路径 (直接回车使用 $assets 文件夹): {Style.RESET\_ALL}")
                if not path:
                    path = "$assets"
                path = os.path.abspath(path)
                if os.path.isfile(path) and sc\_file\_filter(path):
                    process\_file(path, False)
                elif os.path.isdir(path):
                    for name in os.listdir(path):
                        full = os.path.join(path, name)
                        if os.path.isfile(full) and sc\_file\_filter(full):
                            process\_file(full, False)
                else:
                    logger.error("无效的文件或目录路径")

            elif choice == "2":
                path = input(f"{Fore.CYAN}请输入文件或目录路径 (直接回车使用 $assets 文件夹): {Style.RESET\_ALL}")
                if not path:
                    path = "$assets"
                path = os.path.abspath(path)
                if os.path.isfile(path) and sc\_file\_filter(path):
                    process\_file(path, True)
                elif os.path.isdir(path):
                    for name in os.listdir(path):
                        full = os.path.join(path, name)
                        if os.path.isfile(full) and sc\_file\_filter(full):
                            process\_file(full, True)
                else:
                    logger.error("无效的文件或目录路径")

            elif choice == "3":
                logger.info("已启用PNG导出功能")
                dump\_png()
                path = input(f"{Fore.CYAN}请输入文件或目录路径 (直接回车使用 $assets 文件夹): {Style.RESET\_ALL}")
                if not path:
                    path = "$assets"
                path = os.path.abspath(path)
                if os.path.isfile(path) and sc\_file\_filter(path):
                    process\_file(path, False) # 实际实现需结合PNG导出逻辑
                elif os.path.isdir(path):
                    for name in os.listdir(path):
                        full = os.path.join(path, name)
                        if os.path.isfile(full) and sc\_file\_filter(full):
                            process\_file(full, False)
                else:
                    logger.error("无效的文件或目录路径")

            elif choice == "4":
                path = input(f"{Fore.CYAN}请输入需要解压的 .sc 文件路径 (直接回车使用 $assets 文件夹): {Style.RESET\_ALL}")
                if not path:
                    path = "$assets"
                path = os.path.abspath(path)

                if os.path.isfile(path) and path.endswith(".sc"):
                    logger.info(f"正在解压: {os.path.basename(path)}")
                    with open(path, 'rb') as f:
                        raw = f.read().split(b"START")[0]
                    dec = Decompressor().decompress(raw)
                    with open(path + ".dec", 'wb') as out:
                        out.write(dec)
                    logger.info(f"解压完成，保存为: {os.path.basename(path)}.dec")
                elif os.path.isdir(path):
                    logger.info(f"正在处理 {path} 目录下的所有 .sc 文件")
                    for name in os.listdir(path):
                        full = os.path.join(path, name)
                        if os.path.isfile(full) and full.endswith(".sc"):
                            logger.info(f"正在解压: {os.path.basename(full)}")
                            with open(full, 'rb') as f:
                                raw = f.read().split(b"START")[0]
                            dec = Decompressor().decompress(raw)
                            with open(full + ".dec", 'wb') as out:
                                out.write(dec)
                            logger.info(f"解压完成，保存为: {os.path.basename(full)}.dec")
                else:
                    logger.error("请输入有效的 .sc 文件路径或目录")

            elif choice == "5":
                path = input(f"{Fore.CYAN}请输入需要压缩的文件路径 (直接回车使用 $assets 文件夹): {Style.RESET\_ALL}")
                if not path:
                    path = "$assets"
                path = os.path.abspath(path)

                if os.path.isfile(path):
                    logger.info(f"正在压缩: {os.path.basename(path)}")
                    with open(path, 'rb') as f:
                        raw = f.read()
                    cmp = Compressor().compress(raw, Signatures.SC, 1)
                    with open(path + ".cmp", 'wb') as out:
                        out.write(cmp)
                    logger.info(f"压缩完成，保存为: {os.path.basename(path)}.cmp")
                elif os.path.isdir(path):
                    logger.info(f"正在处理 {path} 目录下的所有文件")
                    for name in os.listdir(path):
                        full = os.path.join(path, name)

## 疑难解答

### 为什么闪退？

问：我修改好sc并导入游戏后，要么游戏进不去，要么在放置卡牌时闪退，要么是游戏在游玩过程中闪退？ 答：首先检查An sc工程中的链接是否输入正确，是否和影片剪辑名称一致，有些卡牌图片你没有设置好名称关联会导致进入游戏时游戏找不到卡牌图片而闪退，其次是sc工程中大文件名称是否有误。放置卡牌一般兵种和建筑都是idle1，兵种朝向不能有\_0这一说法，要从1开始（也许），游玩过程中闪退检查你的兵种是否有三类（attack1，run1，idle1），而建筑不需要，建筑只需要idle1就可以了。 新版sc游戏一般都有animations.csv，在这里配置动画文件和名称关联，至于没有animations.csv的老版本，就按上述潜规则进行设置。   问：为什么上面都无误还是会闪退？ 答：这就涉及到兵种属性，比如王子会有冲锋动画，骷髅守卫会有盾牌动画，还有超级骑士的部署动画。这些动画如果你没有创建，但是csv中你的兵种包含了这类属性，就会导致在属性生效时，找不到动画而闪退，正确做法是先将这类特殊兵种建筑的sc转为fla，如果发现纹理有问题，老版本可能sc2fla有bug，就用新版本的游戏sc文件，或使用[ToxicLand/SCEditor: Create or Edit Supercell (Clash of Clans, Clash Royale, Brawl Stars, Boom Beach) SC Files. Add or Edit an Existing or Custom Character, Building or other game object.](https://github.com/ToxicLand/SCEditor) 进行查看，不过这相对麻烦一些，但对于经常使用VS 的人来说挺方便。这个工具还支持在原sc文件中替换图片和添加图片，还支持合并sc，简直棒极了！
